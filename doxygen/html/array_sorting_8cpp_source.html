130void ArraySorting::mergeSubArrays(int subArray[], unsigned int &leftIndex, unsigned int &middleIndex,
131 unsigned int &rightIndex) {

137 int *newSubArray = new int[rightIndex - leftIndex + 1];
138
139 unsigned int startIndexPointer = leftIndex;
